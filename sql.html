Relational Database Basics

A relational database organizes data into tables (rows & columns) and supports relationships between them.

-- Example: Creating a simple table
CREATE TABLE employees (
    id INT PRIMARY KEY,
    name VARCHAR(50),
    salary DECIMAL(10,2)
);

Primary and Foreign Keys

Primary Key

Unique identifier for each row in a table.

CREATE TABLE employees (
    id INT PRIMARY KEY,
    name VARCHAR(50)
);

Foreign Key

Establishes a relationship between two tables.

CREATE TABLE departments (
    dept_id INT PRIMARY KEY,
    emp_id INT,
    FOREIGN KEY (emp_id) REFERENCES employees(id)
);

SQL Operators

Comparison Operators

=, <>, !=, <, >, <=, >=, <=>, IS NULL, IS NOT NULL, IS DISTINCT FROM, ANY, ALL, SOME, EXISTS, BETWEEN, IN

SELECT * FROM employees WHERE salary >= 50000;
SELECT * FROM employees WHERE name IS NOT NULL;

Logical Operators

AND, OR, NOT, XOR

SELECT * FROM employees WHERE salary > 50000 AND dept_id = 1;

Arithmetic Operators

+, -, *, /, %, MOD, DIV

SELECT salary * 0.1 AS bonus FROM employees;

Bitwise Operators

&, |, ^, ~, <<, >>

SELECT id & 1 FROM employees;

String & Search Operators

||, CONCAT, LIKE, NOT LIKE, ILIKE, SIMILAR TO, REGEXP, RLIKE, ESCAPE

SELECT name FROM employees WHERE name LIKE 'J%';

Set Operators

UNION, UNION ALL, INTERSECT, EXCEPT, MINUS

SELECT name FROM employees
UNION
SELECT name FROM contractors;

Clauses & Queries

SELECT & FROM

SELECT name, salary FROM employees;

WHERE

SELECT * FROM employees WHERE salary > 50000;

ORDER BY

SELECT * FROM employees ORDER BY salary DESC;

LIMIT/TOP

SELECT * FROM employees LIMIT 5;

DISTINCT

SELECT DISTINCT dept_id FROM employees;

Subqueries & Correlated Subqueries

SELECT * FROM employees WHERE salary > (SELECT AVG(salary) FROM employees);

Joins

INNER JOIN

SELECT a.name, b.department
FROM employees a
INNER JOIN departments b ON a.id = b.emp_id;

LEFT JOIN

SELECT a.name, b.department
FROM employees a
LEFT JOIN departments b ON a.id = b.emp_id;

RIGHT JOIN

SELECT a.name, b.department
FROM employees a
RIGHT JOIN departments b ON a.id = b.emp_id;

FULL OUTER JOIN

SELECT a.name, b.department
FROM employees a
FULL OUTER JOIN departments b ON a.id = b.emp_id;

CROSS JOIN

SELECT a.name, b.department
FROM employees a CROSS JOIN departments b;

Self Join & Aliases

SELECT e1.name AS Employee, e2.name AS Manager
FROM employees e1
LEFT JOIN employees e2 ON e1.manager_id = e2.id;

SQL Functions

Aggregate Functions

SELECT COUNT(*), SUM(salary), AVG(salary), MIN(salary), MAX(salary) FROM employees;

Group By & Having

SELECT dept_id, AVG(salary) FROM employees GROUP BY dept_id HAVING AVG(salary) > 50000;

String Functions

SELECT CONCAT(name, ' works in ', department) FROM employees;
SELECT LENGTH(name) FROM employees;
SELECT UPPER(name) FROM employees;

Date Functions

SELECT NOW();
SELECT DATE_ADD(NOW(), INTERVAL 7 DAY);

Window Functions

SELECT name, salary, ROW_NUMBER() OVER (PARTITION BY dept_id ORDER BY salary DESC) AS rank
FROM employees;

CASE Statement

SELECT name,
CASE 
    WHEN salary > 50000 THEN 'High'
    ELSE 'Low'
END AS SalaryCategory
FROM employees;

Transactions & ACID Properties

-- Transaction
START TRANSACTION;
INSERT INTO employees (id, name, salary) VALUES (10, 'Alice', 70000);
COMMIT;
-- OR
ROLLBACK;

Advanced Topics

Views

CREATE VIEW high_salary AS
SELECT * FROM employees WHERE salary > 50000;

Stored Procedures

CREATE PROCEDURE GiveBonus(IN empID INT, IN bonusAmount DECIMAL(10,2))
BEGIN
    UPDATE employees SET salary = salary + bonusAmount WHERE id = empID;
END;

Triggers

CREATE TRIGGER after_salary_update
AFTER UPDATE ON employees
FOR EACH ROW
BEGIN
    INSERT INTO audit_log(emp_id, old_salary, new_salary)
    VALUES (OLD.id, OLD.salary, NEW.salary);
END;

User-Defined Functions

CREATE FUNCTION get_tax(salary DECIMAL(10,2)) RETURNS DECIMAL(10,2)
BEGIN
    RETURN salary * 0.1;
END;
